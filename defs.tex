	\@hextodec\R#1#2#1#2%
	\@hextodec\G#3#4#3#4%
	\@hextodec\B#5#6#5#6%
	\edef\@{\endgroup\noexpand\setcolorrgb{\R}{\G}{\B}}%
	\@\ignorespaces}

% 16 shades of grey.
\def\cgrey:#1{%
	\begingroup
	\@hextodec\grey#1#1#1#1%
	\edef\@{\endgroup\noexpand\setcolorgrey{\grey}}%
	\@\ignorespaces}
